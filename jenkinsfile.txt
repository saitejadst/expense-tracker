pipeline {
    agent any

    environment {
        PATH = "${env.PATH}"
    }

    stages {
        stage('Checkout') {
            steps {
                git branch: 'main',
                    url: 'https://github.com/saitejadst/expense-tracker.git',
                    credentialsId: 'github-https'
            }
        }

        stage('Install Dependencies') {
            steps {
                bat 'npm install'
            }
        }

        stage('Run Tests') {
            steps {
                bat 'npm test'
            }
        }
    }

    post {
        success {
            echo 'Build and test successful'
        }
        failure {
            echo 'Build or test failed'
        }
    }
}
