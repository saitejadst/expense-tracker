pipeline {
    agent any

    environment {
        NODEJS_HOME = tool name: 'NodeJS', type: 'NodeJSInstallation'
        PATH = "${env.NODEJS_HOME}/bin:${env.PATH}"
    }

    stages {
        stage('Checkout') {
            steps {
                git branch: 'main', url: 'https://github.com/saitejadst/expense-tracker.git'
            }
        }

        stage('Install Dependencies') {
            steps {
                // Linux/macOS agent
                bat 'npm install'
                // Windows agent: replace with bat 'npm install'
            }
        }

        stage('Run Tests') {
            steps {
                echo 'Running tests'
                // Later, replace echo with actual tests:
                // sh 'npm test'
            }
        }
    }

    post {
        success {
            echo 'Build and test successful '
        }
        failure {
            echo 'Build or test failed '
        }
    }
}
